1) En cuales eventos participa un determinado equipo?

(select evento.nombre from evento,equipo,participan,vs 
 where  evento.id_evento=participan.id_evento_vs 
	and participan.id_evento_vs=vs.id_evento_vs 
	and equipo.nombre='nacional' 
	and equipo.id_equipo=participan.id_equipo)
    union
    (select evento.nombre from evento,equipo,participan_masivo,masivo
    where evento.id_evento=participan_masivo.id_evento_mas and participan_masivo.id_evento_mas=masivo.id_evnt_mas
    and equipo.nombre='nacional'
    and equipo.id_equipo=participan_masivo.id_equipo);

3)Listado de integrantes de un equipo sus datos y a que pais pertenece.

select D.nombre, D.apellido, D.pais,posicion,fecha_fichaje 
from deportista D,equipo E,pertenece P,deportista_group DG 
where D.id_deportista=DG.id_depr_grup
 and DG.id_depr_grup=P.id_depor_grup 
 and E.id_equipo=P.id_equipo
 and E.nombre='nacional';

4)Cuantos partidos a ganado un equipo
(select count(DISTINCT id_evento_vs) 'cantidad de partidos ganados' from participan,equipo
 where participan.id_evento_vs in 
(select vs.id_evento_vs  from vs 
where vs.id_evento_vs in (select id_evento from evento where fecha_inicio between '2022/02/01' and '2022/02/30')) 
and equipo.id_equipo=participan.id_equipo 
and equipo.nombre='Nacional' 
and participan.resultado='gano' );
